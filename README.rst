Environment Variables
---------------------

   # OAUTH
   AUTHLIB_INSECURE_TRANSPORT=
   OAUTH2_JWT_SECRET_KEY=
   OAUTH2_JWT_ISS=
   OAUTH2_JWT_ALG=
   OAUTH2_JWT_EXP=

   # DATABASE
   DB_HOST=
   DB_USER=
   DB_PASSWORD=
   DB_DATABASE=
   DB_PORT=
   DB_ENGINE=

   # MAIL SERVER
   MAIL_SERVER=
   MAIL_PORT=
   MAIL_USE_SSL=
   MAIL_USE_TLS=
   MAIL_USERNAME=
   MAIL_PASSWORD=

   # MAIL SETTING
   MAIL_DEFAULT_SENDER=
   USER_APP_NAME=
   USER_EMAIL_SENDER_EMAIL=

   # SOCILA LOGIN APP
   FACEBOOK_CLIENT_ID=
   FACEBOOK_CLIENT_SECRET=
   GOOGLE_CLIENT_ID=
